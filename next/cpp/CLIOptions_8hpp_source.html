41 std::string to_config(const CLI::App *app, bool default_also, bool, std::string) const override {
42 nlohmann::json j;
43
44 for (const CLI::Option *opt : app->get_options({})) {
45 if (!opt->get_lnames().empty() && opt->get_configurable()) {
46 std::string name = opt->get_lnames()[0];
47
48 if (opt->get_type_size() != 0) {
49 if (opt->count() == 1) j[name] = opt->results().at(0);
50 else if (opt->count() > 1) j[name] = opt->results();
51 else if (default_also && !opt->get_default_str().empty()) j[name] = opt->get_default_str();
52 } else if (opt->count() == 1) {
53 j[name] = true;
54 } else if (opt->count() > 1) {
55 j[name] = opt->count();
56 } else if (opt->count() == 0 && default_also) {
57 j[name] = false;
58 }
59 }
60 }
61
62 for (const CLI::App *subcom : app->get_subcommands({})) {
63 j[subcom->get_name()] = nlohmann::json(to_config(subcom, default_also, false, ""));
64 }
65
66 return j.dump(4);

78 std::vector<CLI::ConfigItem>
79 _from_config(nlohmann::json j, std::string name = "", std::vector<std::string> prefix = {}) const {
80 std::vector<CLI::ConfigItem> results;
81
82 if (j.is_object()) {
83 for (auto item = j.begin(); item != j.end(); ++item) {
84 auto copy_prefix = prefix;
85
86 if (!name.empty()) copy_prefix.push_back(name);
87
88 auto sub_results = _from_config(*item, item.key(), copy_prefix);
89 results.insert(results.end(), sub_results.begin(), sub_results.end());
90 }
91 } else if (!name.empty()) {
92 if (prefix.empty() && !subcommand_names_.empty()) {
93 for (const auto& sub_name : subcommand_names_) {
94 results.emplace_back();
95 CLI::ConfigItem &res = results.back();
96 res.name = name;
97 res.parents = { sub_name };
98 _set_inputs(res, j);
99 }
100 } else {
101 results.emplace_back();
102 CLI::ConfigItem &res = results.back();
103 res.name = name;
104 res.parents = prefix;
105 _set_inputs(res, j);
106 }
107 }
108
109 return results;
110 }

112 void _set_inputs(CLI::ConfigItem &res, const nlohmann::json &j) const {
113 if (j.is_boolean()) {
114 res.inputs = { j.get<bool>() ? "true" : "false" };
115 } else if (j.is_number()) {
116 std::stringstream ss;
117 ss << j.get<double>();
118 res.inputs = { ss.str() };
119 } else if (j.is_string()) {
120 res.inputs = { j.get<std::string>() };
121 } else if (j.is_array()) {
122 for (const auto &val : j) {
123 res.inputs.push_back(val.get<std::string>());
124 }
125 } else {
126 throw CLI::ConversionError("Failed to convert " + res.name);
127 }

181 inline void warn_unused_params(const CLIOptions& params) {
182 if (params.quiet) return;
183
184 if (params.trees == 0) {
185 bool has_warnings = false;
186
187 if (params.threads != -1) {
188 fmt::print("Warning: threads parameter is ignored when training a single tree\n");
189 has_warnings = true;
190 }
191
192 if (params.p_vars != -1 || params.n_vars != -1) {
193 fmt::print("Warning: --vars parameter is ignored when training a single tree\n");
194 has_warnings = true;
195 }
196
197 if (has_warnings) {
198 fmt::print("Single trees always use all features for splitting\n");
199 }

216 inline void init_params(CLIOptions& params, int total_vars = 0) {
217 if (params.lambda == -1) {
218 params.lambda = 0.5;
219 }
220
221 if (params.train_ratio <= 0 || params.train_ratio >= 1) {
222 fmt::print(stderr, "Error: Train ratio must be between 0 and 1\n");
223 exit(1);
224 }
225
226 if (params.seed == -1) {
227 std::random_device rd;
228 params.seed = rd();
229 params.used_default_seed = true;
230 }
231
232 if (params.threads == -1) {
233 #ifdef _OPENMP
234 params.threads = omp_get_max_threads();
235 #else
236 params.threads = 1;
237 #endif
238 params.used_default_threads = true;
239 }
240
241 if (total_vars > 0 && params.trees > 0) {
242 if (params.n_vars != -1) {
243 params.p_vars = static_cast<float>(params.n_vars) / total_vars;
244 } else if (params.p_vars != -1) {
245 params.n_vars = std::round(total_vars * params.p_vars);
246 } else {
247 params.p_vars = 0.5;
248 params.n_vars = std::round(total_vars * params.p_vars);
249 params.used_default_vars = true;
250 }
251 }

266 inline CLIOptions parse_args(int argc, char *argv[]) {
267 CLIOptions params;
268
269 CLI::App app{ "pptree - Projection Pursuit Trees and Forests" };
270 app.require_subcommand(1);
271 app.set_version_flag("--version,-V", PPTREE_VERSION, "Print version and exit");
272
273 // Global options
274 app.add_flag("--quiet,-q", params.quiet, "Suppress all terminal output");
275 app.add_flag("--no-color", params.no_color, "Disable colored output");
276 app.config_formatter(std::make_shared<ConfigJSON>(std::vector<std::string>{ "train", "predict", "evaluate" }));
277 app.set_config("--config", "", "Read parameters from JSON config file");
278
279 // Helper: add model training options shared by train and evaluate
280 auto add_model_options = [&](CLI::App *sub) {
281 sub->add_option("-t,--trees", params.trees, "Number of trees (default: 100, 0 for single tree)")
282 ->check(CLI::NonNegativeNumber);
283 sub->add_option("-l,--lambda", params.lambda, "Method selection (0=LDA, (0,1]=PDA)")
284 ->check(CLI::Range(0.0f, 1.0f));
285 sub->add_option("--threads", params.threads, "Number of threads (default: CPU cores)")
286 ->check(CLI::PositiveNumber);
287 sub->add_option("-r,--seed", params.seed, "Random seed (default: random)");
288 sub->add_option("-v,--vars", params.vars_input, "Features per split (integer=count, decimal or fraction=proportion, default: 0.5)");
289 };
290
291 // Train subcommand
292 auto train_sub = app.add_subcommand("train", "Train a model");
293 train_sub->add_option("-d,--data", params.data_path, "CSV training data")
294 ->required()
295 ->check(CLI::ExistingFile);
296 add_model_options(train_sub);
297 auto train_save_opt = train_sub->add_option("-s,--save", params.save_path, "Save trained model to JSON file (default: model.json)");
298 auto train_no_save = train_sub->add_flag("--no-save", params.no_save, "Skip saving the model (for benchmarking)");
299 train_save_opt->excludes(train_no_save);
300 train_no_save->excludes(train_save_opt);
301 train_sub->add_flag("--no-metrics", params.no_metrics, "Skip variable importance computation and output");
302
303 // Predict subcommand
304 auto predict_sub = app.add_subcommand("predict", "Load a model and predict on new data");
305 predict_sub->add_option("-M,--model", params.model_path, "Saved model JSON file")
306 ->required()
307 ->check(CLI::ExistingFile);
308 predict_sub->add_option("-d,--data", params.data_path, "CSV data to predict on")
309 ->required()
310 ->check(CLI::ExistingFile);
311 predict_sub->add_option("-o,--output", params.output_path, "Save prediction results to JSON file");
312 predict_sub->add_flag("--no-metrics", params.no_metrics, "Omit error rate and confusion matrix from output");
313
314 // Evaluate subcommand
315 auto eval_sub = app.add_subcommand("evaluate", "Train and evaluate a model");
316 auto eval_data_opt = eval_sub->add_option("-d,--data", params.data_path, "CSV file")
317 ->check(CLI::ExistingFile);
318 auto eval_sim_opt = eval_sub->add_option("--simulate", params.simulate, "Simulate NxMxK data");
319 eval_data_opt->excludes(eval_sim_opt);
320 eval_sim_opt->excludes(eval_data_opt);
321
322 add_model_options(eval_sub);
323 eval_sub->add_option("-p,--train-ratio", params.train_ratio, "Train set ratio (default: 0.7)")
324 ->check(CLI::Range(0.01f, 0.99f));
325 eval_sub->add_option("-i,--iterations", params.iterations, "Number of training iterations (default: 1)")
326 ->check(CLI::PositiveNumber);
327 eval_sub->add_option("--sim-mean", params.sim_mean, "Mean for simulated data (default: 100.0)")
328 ->needs(eval_sim_opt);
329 eval_sub->add_option("--sim-mean-separation", params.sim_mean_separation, "Mean separation between classes (default: 50.0)")
330 ->needs(eval_sim_opt)
331 ->check(CLI::PositiveNumber);
332 eval_sub->add_option("--sim-sd", params.sim_sd, "Standard deviation for simulated data (default: 10.0)")
333 ->needs(eval_sim_opt)
334 ->check(CLI::PositiveNumber);
335 eval_sub->add_option("-o,--output", params.output_path, "Save evaluation results to JSON file");
336 eval_sub->add_option("-e,--export", params.export_path, "Export experiment bundle to directory");
337
338 // Parse
339 try {
340 app.parse(argc, argv);
341 } catch (const CLI::ParseError &e) {
342 std::exit(app.exit(e));
343 }
344
345 // Post-parse: determine subcommand
346 if (train_sub->parsed()) {
347 params.subcommand = Subcommand::train;
348 } else if (predict_sub->parsed()) {
349 params.subcommand = Subcommand::predict;
350 } else if (eval_sub->parsed()) {
351 params.subcommand = Subcommand::evaluate;
352 }
353
354 // Post-parse: handle --no-save for train
355 if (params.subcommand == Subcommand::train && params.no_save) {
356 params.save_path.clear();
357 }
358
359 // Post-parse: evaluate has no --save
360 if (params.subcommand == Subcommand::evaluate) {
361 params.save_path.clear();
362 }
363
364 // Post-parse: interpret --vars input
365 if (!params.vars_input.empty()) {
366 try {
367 auto slash_pos = params.vars_input.find('/');
368
369 if (slash_pos != std::string::npos) {
370 int numerator = std::stoi(params.vars_input.substr(0, slash_pos));
371 int denominator = std::stoi(params.vars_input.substr(slash_pos + 1));
372
373 if (denominator <= 0) {
374 fmt::print(stderr, "Error: --vars fraction denominator must be positive\n");
375 std::exit(1);
376 }
377
378 if (numerator <= 0) {
379 fmt::print(stderr, "Error: --vars fraction numerator must be positive\n");
380 std::exit(1);
381 }
382
383 float val = static_cast<float>(numerator) / static_cast<float>(denominator);
384
385 if (val > 1) {
386 fmt::print(stderr, "Error: --vars fraction must evaluate to a proportion between 0 and 1\n");
387 std::exit(1);
388 }
389
390 params.p_vars = val;
391 } else if (params.vars_input.find('.') != std::string::npos) {
392 float val = std::stof(params.vars_input);
393
394 if (val <= 0 || val > 1) {
395 fmt::print(stderr, "Error: --vars proportion must be between 0 and 1\n");
396 std::exit(1);
397 }
398
399 params.p_vars = val;
400 } else {
401 int val = std::stoi(params.vars_input);
402
403 if (val <= 0) {
404 fmt::print(stderr, "Error: --vars count must be positive\n");
405 std::exit(1);
406 }
407
408 params.n_vars = val;
409 }
410 } catch (const std::exception&) {
411 fmt::print(stderr, "Error: Invalid --vars value: {}\n", params.vars_input);
412 std::exit(1);
413 }
414 }
415
416 // Post-parse: validate simulate format
417 if (!params.simulate.empty()) {
418 std::string sim_str = params.simulate;
419 size_t x1 = sim_str.find('x');
420 size_t x2 = sim_str.find('x', x1 + 1);
421
422 if (x1 == std::string::npos || x2 == std::string::npos) {
423 fmt::print(stderr, "Error: Simulate format must be NxMxK (e.g., 1000x10x2)\n");
424 std::exit(1);
425 }
426
427 try {
428 params.rows = std::stoi(sim_str.substr(0, x1));
429 params.cols = std::stoi(sim_str.substr(x1 + 1, x2 - x1 - 1));
430 params.classes = std::stoi(sim_str.substr(x2 + 1));
431
432 if (params.rows <= 0 || params.cols <= 0 || params.classes <= 1) {
433 throw std::out_of_range("Values must be positive and classes must be > 1");
434 }
435 } catch (const std::exception& e) {
436 fmt::print(stderr, "Error: Invalid simulate values: {}\n", e.what());
437 std::exit(1);
438 }
439 }
440
441 // Post-parse: evaluate requires data source
442 if (params.subcommand == Subcommand::evaluate && params.simulate.empty() && params.data_path.empty()) {
443 fmt::print(stderr, "Error: Must specify either --simulate or --data\n");
444 std::exit(1);
445 }
446
447 warn_unused_params(params);
448 return params;
449 }
