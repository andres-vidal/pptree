65inline ProcessResult run_ppforest2(std::string const& args) {
66 ppforest2::io::TempFile stderr_file;
67
68 // clang-format off
69 #ifdef _WIN32
70 std::string cmd = BINARY + " " + args + " 2>\"" + stderr_file.path() + "\"";
71 FILE* pipe = _popen(cmd.c_str(), "r");
72 #else
73 std::string cmd = BINARY + " " + args + " 2>\"" + stderr_file.path() + "\"";
74 FILE* pipe = popen(cmd.c_str(), "r");
75 #endif
76 // clang-format on
77 if (pipe == nullptr) {
78 return {-1, "", ""};
79 }
80
81 std::string output;
82 char buffer[4096];
83
84 while (fgets(buffer, sizeof(buffer), pipe) != nullptr) {
85 output += buffer;
86 }
87
88 // clang-format off
89 #ifdef _WIN32
90 int exit_code = _pclose(pipe);
91 #else
92 int status = pclose(pipe);
93 int exit_code = WIFEXITED(status) ? WEXITSTATUS(status) : -1;
94 #endif
95 // clang-format on
96
97 return {exit_code, output, stderr_file.read()};
98}
