66inline ProcessResult run_ppforest2(const std::string& args) {
67 #ifdef _WIN32
68 std::string cmd = BINARY + " " + args + " 2>NUL";
69 FILE *pipe = _popen(cmd.c_str(), "r");
70 #else
71 std::string cmd = BINARY + " " + args + " 2>/dev/null";
72 FILE *pipe = popen(cmd.c_str(), "r");
73 #endif
74
75 if (!pipe) {
76 return { -1, "" };
77 }
78
79 std::string output;
80 char buffer[4096];
81
82 while (fgets(buffer, sizeof(buffer), pipe))
83 output += buffer;
84
85 #ifdef _WIN32
86 int exit_code = _pclose(pipe);
87 #else
88 int status = pclose(pipe);
89 int exit_code = WIFEXITED(status) ? WEXITSTATUS(status) : -1;
90 #endif
91
92 return { exit_code, output };
93}
